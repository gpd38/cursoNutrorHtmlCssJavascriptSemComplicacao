	- Border-bottom-color
	- Border-left-color
	- Border-color
	- Border-top-style
	- Border-rigth-style
	- Border-bottom-style
	- Border-left-style
	- Border-style (none, hidden, dotted, dashed, solid, double, groove, ridge, inset, outset)
	- Border-top
	- Border-rigth
	- Border-bottom
